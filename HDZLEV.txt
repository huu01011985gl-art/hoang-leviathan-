-- Biến toàn cục
local isHunting = false
local huntThread = nil
local settings = {
    skillKeys = {"Z", "X", "C", "V", "F"},
    teleportDelay = 0.3,
    hangoutPos = CFrame.new(5000, 400, -1500)
}

-- ========== HÀM XỬ LÝ CHÍNH ==========

local function isAlive(char)
    if not char then return false end
    local hum = char:FindFirstChild("Humanoid")
    return hum and hum.Health > 0
end

local function findLeviathan()
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj:IsA("Model") and obj.Name and string.lower(obj.Name):find("leviathan") then
            local hrp = obj:FindFirstChild("HumanoidRootPart")
            local hum = obj:FindFirstChild("Humanoid")
            if hrp and hum and hum.Health > 0 then
                return obj, hrp
            end
        end
    end
    return nil, nil
end

local function pressKey(key)
    local vim = game:GetService("VirtualInputManager")
    pcall(function()
        vim:SendKeyEvent(true, key, false, game)
        task.wait(0.05)
        vim:SendKeyEvent(false, key, false, game)
    end)
end

local function stopHunting()
    if huntThread then
        task.cancel(huntThread)
        huntThread = nil
    end
    local player = game.Players.LocalPlayer
    if player and player.Character then
        local fly = player.Character:FindFirstChild("LeviFly")
        if fly then fly:Destroy() end
    end
end

local function startHunting()
    if huntThread then stopHunting() end
    
    huntThread = task.spawn(function()
        local player = game.Players.LocalPlayer
        
        -- Chờ nhân vật xuất hiện
        repeat
            task.wait(0.5)
            if not isHunting then return end
        until player.Character and player.Character:FindFirstChild("HumanoidRootPart") and isAlive(player.Character)
        
        local hrp = player.Character.HumanoidRootPart
        
        -- Tạo BodyVelocity để bay
        local bodyVel = Instance.new("BodyVelocity")
        bodyVel.Name = "LeviFly"
        bodyVel.Velocity = Vector3.new(0, 0, 0)
        bodyVel.MaxForce = Vector3.new(1e9, 1e9, 1e9)
        bodyVel.Parent = hrp
        
        local lastTele = 0
        
        while isHunting do
            -- Kiểm tra nhân vật
            if not player.Character or not hrp.Parent then
                repeat
                    task.wait(1)
                    if not isHunting then return end
                until player.Character and player.Character:FindFirstChild("HumanoidRootPart")
                hrp = player.Character.HumanoidRootPart
                bodyVel.Parent = hrp
            end
            
            if not isAlive(player.Character) then
                task.wait(3)
                repeat
                    task.wait(1)
                    if not isHunting then return end
                until player.Character and isAlive(player.Character)
                hrp = player.Character.HumanoidRootPart
                bodyVel.Parent = hrp
            end
            
            -- Tìm Leviathan
            local levi, leviHrp = findLeviathan()
            
            if levi and leviHrp then
                local dist = (leviHrp.Position - hrp.Position).Magnitude
                
                -- Tiến đến gần Leviathan
                if dist > 80 then
                    if tick() - lastTele > settings.teleportDelay + 0.2 then
                        hrp.CFrame = leviHrp.CFrame * CFrame.new(0, 35, -25)
                        lastTele = tick()
                    end
                elseif dist > 20 then
                    if tick() - lastTele > settings.teleportDelay then
                        hrp.CFrame = leviHrp.CFrame * CFrame.new(0, 30, -10)
                        lastTele = tick()
                    end
                else
                    -- Trong tầm đánh: spam skill
                    for _, key in pairs(settings.skillKeys) do
                        if not isHunting then break end
                        pressKey(key)
                        task.wait(0.12)
                    end
                    
                    -- Đánh thường
                    local tool = player.Character:FindFirstChildOfClass("Tool")
                    if tool then
                        pcall(function() tool:Activate() end)
                        pcall(function()
                            game:GetService("VirtualUser"):Button1Down(Vector2.new(0,0))
                            task.wait(0.1)
                            game:GetService("VirtualUser"):Button1Up(Vector2.new(0,0))
                        end)
                    end
                end
            else
                -- Không có Levi: về điểm treo
                if tick() - lastTele > 2 then
                    hrp.CFrame = settings.hangoutPos
                    lastTele = tick()
                end
            end
            
            task.wait(0.8)
        end
        
        pcall(function() if bodyVel and bodyVel.Parent then bodyVel:Destroy() end end)
    end)
end

-- Cập nhật CanvasSize
local function updateCanvas()
    local totalHeight = 0
    for _, child in pairs(scrollFrame:GetChildren()) do
        if child:IsA("Frame") and child ~= listLayout then
            totalHeight = totalHeight + child.Size.Y.Offset + 12
        end
    end
    scrollFrame.CanvasSize = UDim2.new(0, 0, 0, totalHeight + 20)
end
